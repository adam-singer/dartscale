Mediator class 

class Mediator {
 
 var _channels = new Map<String, MediatorChannel>();
 
 MediatorChannel operator [](String channelName) {
   var channel = _channels[channelName];
   if (channel == null) {
     channel = new MediatorChannel();
     _channels[channelName] = channel;
   }
   
   return channel;
 }
}
