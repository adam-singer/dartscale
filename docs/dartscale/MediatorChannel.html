class MediatorChannel {
 
 var _topics = new Map<String, StreamController>();
 
 StreamController operator [](String topicName) {
   StreamController stream = _topics[topicName];
   if (stream == null) {
     stream = new StreamController();
     _topics[topicName] = stream;
   }
   
   return stream;
 }
}
